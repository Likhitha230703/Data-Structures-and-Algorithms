import java.util.ArrayList;

public class Solution 
{
    public static int findLargestMinDistance(ArrayList<Integer> boards, int k)
    {
        //    Write your code here.
        int max=Integer.MIN_VALUE,sum=0;
        for(int i=0;i<boards.size();i++){
            if(boards.get(i)>max){
                max=boards.get(i);
            }
            sum+=boards.get(i);
        }
        int low=max,high=sum;
        while(low<=high){
            int mid=(low+high)/2;
            int ans=findarea(boards,mid);
            if(ans<=k){
                high=mid-1;
            }
            else{
                low=mid+1;
            }
        }
        return low;
    }
    public static int findarea(ArrayList<Integer> boards,int mid){
        int area=0,parts=1;
        for(int i=0;i<boards.size();i++){
            area+=boards.get(i);
            if(area>mid){
                area=boards.get(i);
                parts++;
            }
        }
        return parts;
    }
}
